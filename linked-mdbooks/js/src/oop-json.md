# JSON

### Example
```js
var myMusic = [               // myMusic is an Array with objects! 
  {                           // {} -> Object
    "artist": "Billy Joel",   // property-value - pairs
    "title": "Piano Man",
    "release_year": 1973,
    "formats": [              // value can also be an array
      "CD",
      "8T",
      "LP"
    ],
    "gold": true
  },                          // objects are seperated with comma
  {
    "artist": "Billy Joel",
    "title": "Piano Man",
    "release_year": 1973,
    "formats": [ 
      "CD",
      "8T",
      "LP"
    ],
    "gold": true
  }
];
```
